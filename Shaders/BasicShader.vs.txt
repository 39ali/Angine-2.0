#version 400 core

layout(location=0) in vec3 pos ;
layout(location=1) in vec3 normal;
layout(location=2) in vec2 uv;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection=mat4(1.0f);

out vec2 o_uv;
void main ()
{
gl_Position = projection*view*model *vec4(pos, 1.0);
o_uv= vec2(uv.x,1-uv.y);
}
